00029  * FindB 
00030  * @brief Find B to run CLCG4 backwards
00031  * 
00032  * B is \f$ a_[i]^{m_[i] - 2} \mathrm{mod m_[i]} \f$                   
00033  * which is used in running the CLCG4 backwards                        
00034  * Added by Chris Carothers, 5/15/98                          
00035  */
00036 long long

00038 {
00039   long long sqrs[32];
00040   long long power_of_2;
00041   long long b;
00042 
00043   int i;
00044 
00045   sqrs[0] = a;
00046   for(i = 1; i < 32; i++)
00047     {
00048       sqrs[i] =(sqrs[i - 1] * sqrs[i - 1]) % m;
00049     }
00050 
00051   power_of_2 = 1;
00052   b = 1;
00053   for(i = 0; i < 32; i++)
00054     {
00055       if(!(power_of_2 & k))
00056         {
00057           sqrs[i] = 1;
00058 
00059         }
00060       b =(b * sqrs[i]) % m;
00061       power_of_2 = power_of_2 * 2;
00062     }
00063 
00064   return(b);

00099           S1 -= H;

00116           while(R < 0)
