00005  *  post-reverse - reset the bitfield so that a potential re-running of the
00006  *  event is presented with a consistent bitfield state
00007  *  NOTE: the size checks are to better support the experimental reverse
00008  *  computation compiler, which can use a larger bitfield.

00012 {
00013     if (sizeof(revent->cv) == sizeof(uint32_t)){
00014         *(uint32_t*)&revent->cv = 0;

00016     else if (sizeof(revent->cv) == sizeof(uint64_t)){
00017         *(uint64_t*)&revent->cv = 0;
00018     }
00019     else{
00020         memset(&revent->cv, 0, sizeof(revent->cv));
