00058 static void sift_down( tw_pq *h, int i )
00059 {
00060   int n = h->nelems, k = i, j, c1, c2;
00061   ELEMENT_TYPE temp;
00062 
00063   if( n <= 1 ) return;
00064 
00065   /* Stops when neither child is "strictly less than" parent */
00066   do{
00067     j = k;
00068     c1 = c2 = 2*k+1;
00069     c2++;
00070     if( c1 < n && KEY(h->elems[c1]) < KEY(h->elems[k]) ) k = c1;
00071     if( c2 < n && KEY(h->elems[c2]) < KEY(h->elems[k]) ) k = c2;
00072     SWAP( h, j, k, temp );
00073   }while( j != k );
00074 }

00077 static void percolate_up( tw_pq *h, int i )
00078 {
00079   int n = h->nelems, k = i, j, p;
00080   ELEMENT_TYPE temp;
00081 
00082   if( n <= 1 ) return;
00083 
00084   /* Stops when parent is "less than or equal to" child */
00085   do
00086     {
00087       j = k;
00088       if( (p = (k+1)/2) )
00089         {
00090           --p;
00091           if( KEY(h->elems[k]) < KEY(h->elems[p]) ) k = p;
00092         }
00093       SWAP( h, j, k, temp );
